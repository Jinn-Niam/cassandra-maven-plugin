64      /**
65       * Stops the Cassandra service.
66       *
67       * @param rpcAddress The rpcAddress to connect to in order to see if Cassandra has stopped.
68       * @param rpcPort    The rpcPort to connect on to check if Cassandra has stopped.
69       * @param stopPort   The port to stop on.
70       * @param stopKey    The key to stop with,
71       * @param log        The log to write to.
72       */
73      static void stopCassandraServer(String rpcAddress, int rpcPort, String stopAddress, int stopPort, String stopKey, Log log)
74      {
75          try
76          {
77              Socket s = new Socket(InetAddress.getByName(stopAddress), stopPort);
78              s.setSoLinger(false, 0);
79  
80              OutputStream out = s.getOutputStream();
81              out.write((stopKey + "\r\nstop\r\n").getBytes());
82              out.flush();
83              s.close();
84          } catch (ConnectException e)
85          {
86              log.info("Cassandra not running!");
87              return;
88          } catch (Exception e)
89          {
90              log.error(e);
91              return;
92          }
93          log.info("Waiting for Cassandra to stop...");
94          long maxWaiting = System.currentTimeMillis() + TimeUnit.SECONDS.toMillis(30);
95          boolean stopped = false;
96          while (!stopped && System.currentTimeMillis() < maxWaiting)
97          {
98              TTransport tr = new TFramedTransport(new TSocket(rpcAddress, rpcPort));
99              try
100             {
101                 TProtocol proto = new TBinaryProtocol(tr);
102                 Cassandra.Client client = new Cassandra.Client(proto);
103                 try
104                 {
105                     tr.open();
106                 } catch (TTransportException e)
107                 {
108                     if (e.getCause() instanceof ConnectException)
109                     {
110                         stopped = true;
111                         continue;
112                     }
113                     log.debug(e.getLocalizedMessage(), e);
114                     try
115                     {
116                         Thread.sleep(500);
117                     } catch (InterruptedException e1)
118                     {
119                         // ignore
120                     }
121                 }
122             } finally
123             {
124                 if (tr.isOpen())
125                 {
126                     tr.close();
127                 }
128             }
129         }
130         if (stopped)
131         {
132             log.info("Cassandra has stopped.");
133         } else
134         {
135             log.warn("Gave up waiting for Cassandra to stop.");
136         }
137     }

139     /**
140      * Starts the Cassandra server.
141      *
142      * @param cassandraDir The directory to start the Server process in.
143      * @param commandLine  The command line to use to start the Server process.
144      * @param environment  The environment to start the Server process with.
145      * @param log          The log to send the output to.
146      * @return The {@link ExecuteResultHandler} for the started process.
147      * @throws MojoExecutionException if something went wrong.
148      */
149     protected static DefaultExecuteResultHandler startCassandraServer(File cassandraDir, CommandLine commandLine,
150                                                                       Map environment, Log log)
151             throws MojoExecutionException
152     {
153 
154         try
155         {
156             Executor exec = new DefaultExecutor();
157             DefaultExecuteResultHandler execHandler = new DefaultExecuteResultHandler();
158             exec.setWorkingDirectory(cassandraDir);
159             exec.setProcessDestroyer(new ShutdownHookProcessDestroyer());
160 
161             LogOutputStream stdout = new MavenLogOutputStream(log);
162             LogOutputStream stderr = new MavenLogOutputStream(log);
163 
164             log.debug("Executing command line: " + commandLine);
165 
166             exec.setStreamHandler(new PumpStreamHandler(stdout, stderr));
167 
168             exec.execute(commandLine, environment, execHandler);
169 
170             return execHandler;
171         } catch (ExecuteException e)
172         {
173             throw new MojoExecutionException("Command execution failed.", e);
174         } catch (IOException e)
175         {
176             throw new MojoExecutionException("Command execution failed.", e);
177         }
178     }

180     /**
181      * Returns {@code true} if the resource is not a file, does not exist or is older than the project file.
182      *
183      * @param project  the project that the resource is dependent on.
184      * @param resource the resource to query.
185      * @return {@code true} if the resource is not a file, does not exist or is older than the project file.
186      */
187     static boolean shouldGenerateResource(MavenProject project, File resource)
188     {
189         if (!resource.isFile())
190         {
191             return true;
192         }
193         long resourceLM = resource.lastModified();
194         long projectLM = project.getFile().lastModified();
195         if (Long.signum(resourceLM) == Long.signum(projectLM))
196         {
197             // the two dates are in the same epoch or else the universe is lasting a really long time.
198             return resourceLM < projectLM;
199         }
200         // the universe has been around long enough that we should rewrite the resource.
201         return true;
202     }

204     /**
205      * Applies the glossYaml on top of the baseYaml and returns the result.
206      *
207      * @param baseYaml  the base Yaml.
208      * @param glossYaml the Yaml to overide the base with.
209      * @return the resulting Yaml.
210      */
211     public static String merge(String baseYaml, String glossYaml)
212     {
213         if (StringUtils.isBlank(glossYaml))
214         {
215             return baseYaml;
216         }
217         if (StringUtils.isBlank(baseYaml))
218         {
219             return glossYaml;
220         }
221         Yaml yaml = new Yaml();
222         Map<String, Object> baseMap = (Map<String, Object>) yaml.load(baseYaml);
223         Map<String, Object> glossMap = (Map<String, Object>) yaml.load(glossYaml);
224         for (Map.Entry<String, Object> glossEntry : glossMap.entrySet())
225         {
226             baseMap.put(glossEntry.getKey(), glossEntry.getValue());
227         }
228         return yaml.dump(baseMap);
229     }

231     /**
232      * Waits until the Cassandra server at the specified RPC address and port has started accepting connections.
233      *
234      * @param rpcAddress       The RPC address to connect to.
235      * @param rpcPort          The RPC port to connect on.
236      * @param startWaitSeconds The maximum number of seconds to wait.
237      * @param log              the {@link Log} to log to.
238      * @return {@code true} if Cassandra is started.
239      * @throws MojoExecutionException if something went wrong.
240      */
241     static boolean waitUntilStarted(String rpcAddress, int rpcPort, int startWaitSeconds, Log log)
242             throws MojoExecutionException
243     {
244         long maxWaiting = System.currentTimeMillis() + TimeUnit.SECONDS.toMillis(startWaitSeconds);
245         while (startWaitSeconds == 0 || System.currentTimeMillis() < maxWaiting)
246         {
247             TTransport tr = new TFramedTransport(new TSocket(rpcAddress, rpcPort));
248             try
249             {
250                 TProtocol proto = new TBinaryProtocol(tr);
251                 Cassandra.Client client = new Cassandra.Client(proto);
252                 try
253                 {
254                     tr.open();
255                 } catch (TTransportException e)
256                 {
257                     if (!(e.getCause() instanceof ConnectException))
258                     {
259                         log.debug(e.getLocalizedMessage(), e);
260                     }
261                     try
262                     {
263                         Thread.sleep(500);
264                     } catch (InterruptedException e1)
265                     {
266                         // ignore
267                     }
268                     continue;
269                 }
270                 try
271                 {
272                     log.info("Cassandra cluster \"" + client.describe_cluster_name() + "\" started.");
273                     return true;
274                 } catch (TException e)
275                 {
276                     throw new MojoExecutionException(e.getLocalizedMessage(), e);
277                 }
278             } finally
279             {
280                 if (tr.isOpen())
281                 {
282                     tr.close();
283                 }
284             }
285         }
286         return false;
287     }

289     /**
290      * Call {@link #executeOperation(Cassandra.Client)} on the provided operation
291      * @throws MojoExecutionException
292      * @throws MojoFailureException
293      */    
294     public static void executeThrift(ThriftApiOperation thriftApiOperation) throws MojoExecutionException
295     {
296         TSocket socket = new TSocket(thriftApiOperation.getRpcAddress(), thriftApiOperation.getRpcPort());
297         TTransport transport = new TFramedTransport(socket);
298 
299         TBinaryProtocol binaryProtocol = new TBinaryProtocol(transport, true, true);
300         Cassandra.Client cassandraClient = new Cassandra.Client(binaryProtocol);
301 
302         try 
303         {
304             transport.open();
305             if ( StringUtils.isNotBlank(thriftApiOperation.getKeyspace()) ) 
306             {
307                 cassandraClient.set_keyspace(thriftApiOperation.getKeyspace());
308             }
309             cassandraClient.set_cql_version(thriftApiOperation.getCqlVersion());
310             thriftApiOperation.executeOperation(cassandraClient);
311         } catch (ThriftApiExecutionException taee) 
312         {
313             throw new MojoExecutionException("API Exception calling Apache Cassandra", taee);
314         } catch (Exception ex) 
315         {
316             throw new MojoExecutionException("General exception from Thrift", ex);
317         }
318         
319         finally 
320         {
321             if ( transport != null && transport.isOpen() )
322             {
323                 try 
324                 {
325                     transport.flush();
326                     transport.close();
327                 } catch (Exception e) 
328                 { 
329                     throw new MojoExecutionException("Something went wrong cleaning up", e);
330                 }
331             }
332         }
