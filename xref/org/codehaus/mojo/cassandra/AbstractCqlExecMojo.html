22  /**
23   * Abstract parent class for mojos that need to run CQL statements.
24   *
25   * @author sparhomenko
26   */
27  public abstract class AbstractCqlExecMojo extends AbstractCassandraMojo
28  {
29      /**
30       * Version of CQL to use
31       *
32       * @parameter property="cql.version"
33       * @since 1.2.1-2
34       */
35      private String cqlVersion = "3.4.0";
36  
37      /**
38       * Charset used when loading CQL files. If not specified the system default encoding will be used.
39       *
40       * @parameter property="cql.encoding"
41       * @since 3.6
42       */
43      protected String cqlEncoding = Charset.defaultCharset().name();
44  
45      protected String readFile(File file) throws MojoExecutionException
46      {
47          if (!file.isFile())
48          {
49              throw new MojoExecutionException("script " + file + " does not exist.");
50          }
51  
52          InputStreamReader r = null;
53          try
54          {
55              r = new InputStreamReader(new FileInputStream(file), cqlEncoding);
56              return IOUtil.toString(r);
57          } catch (FileNotFoundException e)
58          {
59              throw new MojoExecutionException("Cql file '" + file + "' was deleted before I could read it", e);
60          } catch (IOException e)
61          {
62              throw new MojoExecutionException("Could not parse or load cql file", e);
63          } finally
64          {
65              IOUtil.close(r);
66          }
67      }
68  
69      protected List<CqlResult> executeCql(final String statements) throws MojoExecutionException
70      {
71          final List<CqlResult> results = new ArrayList<CqlResult>();
72          if (StringUtils.isBlank(statements))
73          {
74              getLog().warn("No CQL provided. Nothing to do.");
75          } else
76          {
77              try
78              {
79                  CqlExecOperation operation = new CqlExecOperation(statements);
80                  Utils.executeThrift(operation);
81                  results.addAll(operation.results);
82              } catch (ThriftApiExecutionException taee)
83              {
84                  throw new MojoExecutionException(taee.getMessage(), taee);
85              }
86          }
87          return results;
88      }
89  
90      private class CqlExecOperation extends ThriftApiOperation
91      {
92          private final List<CqlResult> results = new ArrayList<CqlResult>();
93          private final String[] statements;
94  
95          private CqlExecOperation(String statements)
96          {
97              super(rpcAddress, rpcPort);
98              this.statements = statements.split(";");
99              if (StringUtils.isNotBlank(keyspace))
100             {
101                 getLog().info("setting keyspace: " + keyspace);
102                 setKeyspace(keyspace);
103             }
104             getLog().info("setting cqlversion: " + cqlVersion);
105             setCqlVersion(cqlVersion);
106         }
107 
108         @Override
109         void executeOperation(Client client) throws ThriftApiExecutionException
110         {
111             for (String statement : statements)
112             {
113                 if (StringUtils.isNotBlank(statement))
114                 {
115                     results.add(executeStatement(client, statement));
116                 }
117             }
118         }
119 
120         private CqlResult executeStatement(Client client, String statement) throws ThriftApiExecutionException
121         {
122             ByteBuffer buf = ByteBufferUtil.bytes(statement);
123             try
124             {
125                 if (cqlVersion.charAt(0) >= '3')
126                 {
127                     return client.execute_cql3_query(buf, Compression.NONE, ConsistencyLevel.ONE);
128                 } else
129                 {
130                     return client.execute_cql_query(buf, Compression.NONE);
131                 }
132             } catch (Exception e)
133             {
134                 getLog().debug(statement);
135                 throw new ThriftApiExecutionException(e);
136             }
137         }
138     }
